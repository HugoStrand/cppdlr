68 template <nda::MemoryArrayOfRank<2> T, typename S = get_value_t<T>>
69 requires(nda::is_scalar_v<S>)
70 std::tuple<typename T::regular_type, nda::vector<double>, nda::vector<int>> pivrgs(T const &a, double eps) {
71
72 auto _ = nda::range::all;
73
74 // Copy input data
75 auto aa = make_regular(a);
76
77 // Get matrix dimensions
78 auto [m, n] = aa.shape();
79 int maxrnk = std::min(m, n);
80
81 // Compute norms of rows of input matrix, and rescale eps tolerance
82 auto norms = nda::vector<double>(m);
83 double epsscal = 0; // Scaled eps threshold parameter
84 for (int j = 0; j < m; ++j) {
85 norms(j) = real(blas::dotc(aa(j, _), aa(j, _)));
86 // TODO: Need to choose between these; this choice is consistent w/ libdlr
87 //epsscal += norms(j);
88 epsscal = std::max(epsscal, norms(j));
89 }
90 epsscal *= eps * eps;
91
92 // Begin pivoted double Gram-Schmidt procedure
93 int jpiv = 0, jj = 0;
94 double nrm = 0;
95 auto piv = nda::arange(m);
96 auto tmp = nda::vector<S>(n);
97
98 for (int j = 0; j < maxrnk; ++j) {
99
100 // Find next pivot
101 jpiv = j;
102 nrm = norms(j);
103 for (int k = j + 1; k < m; ++k) {
104 if (norms(k) > nrm) {
105 jpiv = k;
106 nrm = norms(k);
107 }
108 }
109
110 // Swap current row with chosen pivot row
111 tmp = aa(j, _);
112 aa(j, _) = aa(jpiv, _);
113 aa(jpiv, _) = tmp;
114
115 nrm = norms(j);
116 norms(j) = norms(jpiv);
117 norms(jpiv) = nrm;
118
119 jj = piv(j);
120 piv(j) = piv(jpiv);
121 piv(jpiv) = jj;
122
123 // Orthogonalize current rows (now the chosen pivot row) against all
124 // previously chosen rows
125 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * blas::dotc(aa(k, _), aa(j, _)); }
126
127 // Get norm of current row
128 nrm = real(blas::dotc(aa(j, _), aa(j, _)));
129 //nrm = nda::norm(aa(j, _));
130
131 // Terminate if sufficiently small, and return previously selected rows
132 // (not including current row)
133 if (nrm <= epsscal) { return {aa(nda::range(j), _), norms(nda::range(j)), piv(nda::range(j))}; };
134
135 // Normalize current row
136 aa(j, _) = aa(j, _) * (1 / sqrt(nrm));
137
138 // Orthogonalize remaining rows against current row
139 for (int k = j + 1; k < m; ++k) {
140 if (norms(k) <= epsscal) { continue; } // Can skip rows with norm less than tolerance
141 aa(k, _) = aa(k, _) - aa(j, _) * blas::dotc(aa(j, _), aa(k, _));
142 norms(k) = real(blas::dotc(aa(k, _), aa(k, _)));
143 }
144 }
145
146 return {aa(nda::range(maxrnk), _), norms(nda::range(maxrnk)), piv(nda::range(maxrnk))};
147 }

229 template <nda::MemoryArray Ta, nda::MemoryArray Tb, nda::Scalar Sa = nda::get_value_t<Ta>, nda::Scalar Sb = nda::get_value_t<Tb>,
230 nda::Scalar S = typename std::common_type<Sa, Sb>::type>
231 nda::array<S, Ta::rank + Tb::rank - 2> arraymult(Ta const &a, Tb const &b) {
232
233 // Get ranks of input arrays
234 constexpr int ra = Ta::rank;
235 constexpr int rb = Tb::rank;
236
237 // Get inner dimensions of input arrays
238 int p = a.shape(ra - 1);
239 if (b.shape(0) != p) throw std::runtime_error("last dim of a != first dim of b");
240
241 // Get product of outer dimensions of input arrays
242 int m = a.size() / p;
243 int n = b.size() / p;
244
245 // Reshape input arrays to 2D arrays
246 auto a_reshaped = nda::reshape(a, m, p);
247 auto b_reshaped = nda::reshape(b, p, n);
248
249 // Get shape of output array
250 auto c_shape = std::array<int, ra + rb - 2>();
251 for (int i = 0; i < ra - 1; ++i) { c_shape[i] = a.shape(i); }
252 for (int i = ra - 1; i < ra + rb - 2; ++i) { c_shape[i] = b.shape(i - ra + 2); }
253
254 // Compute the contraction, reshape, and return
255 return reshape(matmul(a_reshaped, b_reshaped), c_shape);
256 }

281 S adapgl(std::function<nda::array<S, 1>(nda::array<double, 1>)> f, double a, double b, double tol, nda::vector<double> xgl,
282 nda::vector<double> wgl) {
283
284 int maxnde = 100000; // Max nodes in integration tree
285 int maxstk = 1000; // Max stack size
286
287 S sum = 0; // Integral
288
289 // Initialize stack: there is a stack of interval endpoints associated w/
290 // each node in the integration tree, and a corresponding stack of values
291 // associated w/ each such interval
292
293 auto endpt = nda::array<double, 2>(maxstk, 2); // Stack of interval endpoints
294 auto sums = nda::array<S, 1>(maxstk); // Stack of sums associated w/ each interval
295
296 int istk = 0; // Stack pointer
297 int nnde = 1; // Number of nodes in integration tree
298
299 endpt(istk, 0) = a; // Root node interval is [a,b]
300 endpt(istk, 1) = b;
301
302 double h = (b - a) / 2; // Interval half-width
303 sums(istk) = h * nda::blas::dotc(wgl, f(xgl * h + (a + b) / 2)); // Integral of f on root node interval
304
305 double aa = 0, bb = 0, cc = 0; // Interval endpoints, midpoint
306 S sum1 = 0, sum2 = 0; // Integrals on subintervals
307
308 // Main loop
309 while (true) {
310
311 aa = endpt(istk, 0); // Get interval endpoints
312 bb = endpt(istk, 1);
313 cc = (aa + bb) / 2; // Get midpoint
314
315 // Get integral on each subinterval
316 h = (cc - aa) / 2;
317 sum1 = h * nda::blas::dotc(wgl, f(xgl * h + (aa + cc) / 2));
318 sum2 = h * nda::blas::dotc(wgl, f(xgl * h + (cc + bb) / 2));
319
320 // If error on subintervals is less than tolerance, add to integral and
321 // move to next node
322
323 if (std::abs(sum1 + sum2 - sums(istk)) < tol) {
324 sum += sum1 + sum2; // Add to subintervals to integral
325 istk--; // Remove current interval from stack
326 if (istk < 0) { return sum; } // If stack is empty, return integral
327 } else { // Integral on current interval not converged
328
329 // Remove current interval from stack, and add subintervals
330 endpt(istk, 1) = cc;
331 endpt(istk + 1, 0) = cc;
332 endpt(istk + 1, 1) = bb;
333 sums(istk) = sum1;
334 sums(istk + 1) = sum2;
335
336 istk++;
337 nnde += 2;
338
339 if (istk > maxstk) { throw std::runtime_error("integration stack too large"); }
340 if (nnde > maxnde) { throw std::runtime_error("integration tree too large"); }
341 }
342 }
343 }
