77 
78  // Get matrix dimensions
79  auto [m, n] = aa.shape();
80  int maxrnk = std::min(m, n);
81 
82  // Compute norms of rows of input matrix, and rescale eps tolerance
83  auto norms = nda::vector<double>(m);
84  double epsscal = 0; // Scaled eps threshold parameter
85  for (int j = 0; j < m; ++j) {
86  norms(j) = real(blas::dotc(aa(j, _), aa(j, _)));
87  // TODO: Need to choose between these; this choice is consistent w/ libdlr
88  //epsscal += norms(j);
89  epsscal = std::max(epsscal, norms(j));
90  }
91  epsscal *= eps * eps;
92 
93  // Begin pivoted double Gram-Schmidt procedure
94  int jpiv = 0, jj = 0;
95  double nrm = 0;
96  auto piv = nda::arange(m);
97  auto tmp = nda::vector<S>(n);
98 
99  for (int j = 0; j < maxrnk; ++j) {
100 
101  // Find next pivot
102  jpiv = j;
103  nrm = norms(j);
104  for (int k = j + 1; k < m; ++k) {
105  if (norms(k) > nrm) {
106  jpiv = k;
107  nrm = norms(k);
108  }
109  }
110 
111  // Swap current row with chosen pivot row
112  tmp = aa(j, _);
113  aa(j, _) = aa(jpiv, _);
114  aa(jpiv, _) = tmp;
115 
116  nrm = norms(j);
117  norms(j) = norms(jpiv);
118  norms(jpiv) = nrm;
119 
120  jj = piv(j);
121  piv(j) = piv(jpiv);
122  piv(jpiv) = jj;
123 
124  // Orthogonalize current rows (now the chosen pivot row) against all
125  // previously chosen rows
126  for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * blas::dotc(aa(k, _), aa(j, _)); }
127 
128  // Get norm of current row
129  nrm = real(blas::dotc(aa(j, _), aa(j, _)));
130  //nrm = nda::norm(aa(j, _));
131 
132  // Terminate if sufficiently small, and return previously selected rows
133  // (not including current row)
134  if (nrm <= epsscal) { return {aa(nda::range(0, j), _), norms(nda::range(0, j)), piv(nda::range(0, j))}; };
135 
136  // Normalize current row
137  aa(j, _) = aa(j, _) * (1 / sqrt(nrm));
138 
139  // Orthogonalize remaining rows against current row
140  for (int k = j + 1; k < m; ++k) {
141  if (norms(k) <= epsscal) { continue; } // Can skip rows with norm less than tolerance
142  aa(k, _) = aa(k, _) - aa(j, _) * blas::dotc(aa(j, _), aa(k, _));
143  norms(k) = real(blas::dotc(aa(k, _), aa(k, _)));
144  }

540  nda::Scalar S = typename std::common_type<Sa, Sb>::type>
541  nda::array<S, Ta::rank + Tb::rank - 2> arraymult(Ta const &a, Tb const &b) {
542 
543  // Get ranks of input arrays
544  constexpr int ra = Ta::rank;
545  constexpr int rb = Tb::rank;
546 
547  // Get inner dimensions of input arrays
548  int p = a.shape(ra - 1);
549  if (b.shape(0) != p) throw std::runtime_error("last dim of a != first dim of b");
550 
551  // Get product of outer dimensions of input arrays
552  int m = a.size() / p;
553  int n = b.size() / p;
554 
555  // Reshape input arrays to 2D arrays
556  auto a_reshaped = nda::reshape(a, m, p);
557  auto b_reshaped = nda::reshape(b, p, n);
558 
559  // Get shape of output array
560  auto c_shape = std::array<int, ra + rb - 2>();
561  for (int i = 0; i < ra - 1; ++i) { c_shape[i] = a.shape(i); }
562  for (int i = ra - 1; i < ra + rb - 2; ++i) { c_shape[i] = b.shape(i - ra + 2); }
563 
564  // Compute the contraction, reshape, and return
565  return reshape(matmul(a_reshaped, b_reshaped), c_shape);

589  template <typename S>
590  // TODO: require S is scalar
591  S adapgl(std::function<nda::array<S, 1>(nda::array<double, 1>)> f, double a, double b, double tol, nda::vector<double> xgl,
592  nda::vector<double> wgl) {
593 
594  int maxnde = 100000; // Max nodes in integration tree
595  int maxstk = 1000; // Max stack size
596 
597  S sum = 0; // Integral
598 
599  // Initialize stack: there is a stack of interval endpoints associated w/
600  // each node in the integration tree, and a corresponding stack of values
601  // associated w/ each such interval
602 
603  auto endpt = nda::array<double, 2>(maxstk, 2); // Stack of interval endpoints
604  auto sums = nda::array<S, 1>(maxstk); // Stack of sums associated w/ each interval
605 
606  int istk = 0; // Stack pointer
607  int nnde = 1; // Number of nodes in integration tree
608 
609  endpt(istk, 0) = a; // Root node interval is [a,b]
610  endpt(istk, 1) = b;
611 
612  double h = (b - a) / 2; // Interval half-width
613  sums(istk) = h * nda::blas::dotc(wgl, f(xgl * h + (a + b) / 2)); // Integral of f on root node interval
614 
615  double aa = 0, bb = 0, cc = 0; // Interval endpoints, midpoint
616  S sum1 = 0, sum2 = 0; // Integrals on subintervals
617 
618  // Main loop
619  while (true) {
620 
621  aa = endpt(istk, 0); // Get interval endpoints
622  bb = endpt(istk, 1);
623  cc = (aa + bb) / 2; // Get midpoint
624 
625  // Get integral on each subinterval
626  h = (cc - aa) / 2;
627  sum1 = h * nda::blas::dotc(wgl, f(xgl * h + (aa + cc) / 2));
628  sum2 = h * nda::blas::dotc(wgl, f(xgl * h + (cc + bb) / 2));
629 
630  // If error on subintervals is less than tolerance, add to integral and
631  // move to next node
632 
633  if (std::abs(sum1 + sum2 - sums(istk)) < tol) {
634  sum += sum1 + sum2; // Add to subintervals to integral
635  istk--; // Remove current interval from stack
636  if (istk < 0) { return sum; } // If stack is empty, return integral
637  } else { // Integral on current interval not converged
638 
639  // Remove current interval from stack, and add subintervals
640  endpt(istk, 1) = cc;
641  endpt(istk + 1, 0) = cc;
642  endpt(istk + 1, 1) = bb;
643  sums(istk) = sum1;
644  sums(istk + 1) = sum2;
645 
646  istk++;
647  nnde += 2;
648 
649  if (istk > maxstk) { throw std::runtime_error("integration stack too large"); }
650  if (nnde > maxnde) { throw std::runtime_error("integration tree too large"); }
651  }
652  }
653  }
